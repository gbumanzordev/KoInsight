    1. Add POST route to `apps/server/src/books/books-router.ts`:
       ```typescript
       import { enrichmentService } from '../enrichment/service';
       import { db } from '../knex';
       // ...
       router.post('/:bookId/re-enrich', getBookById, async (req: Request, res: Response) => {
         const book = req.book!;
         try {
           await enrichmentService.enqueue(book.md5);
           // Pitfall 5: prefer the open job; fall back to the most recent terminal row.
           const openJob = await db('enrichment_job')
             .where({ book_md5: book.md5 })
             .whereIn('status', ['pending', 'running'])
             .orderBy('id', 'desc')
             .first();
           const job = openJob
             ?? (await db('enrichment_job').where({ book_md5: book.md5 }).orderBy('id', 'desc').first())
             ?? null;
           res.status(202).json({ job });
         } catch (error) {
           console.error(error);
           res.status(500).json({ error: 'Failed to enqueue re-enrichment' });
         }
       });
       ```
