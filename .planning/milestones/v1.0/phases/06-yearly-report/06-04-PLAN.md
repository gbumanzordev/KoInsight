```typescript
// Generic bucket-with-Unknown: groups raw {md5, value} rows by value, NULL -> 'Unknown'.
// counter='per-book' counts distinct md5s per value (nationality, language).
// counter='per-row' counts every row (genre, each book contributes one count per genre).
export function bucketWithUnknown(
  rows: Array<{ md5: string; value: string | null }>,
  counter: 'per-book' | 'per-row'
): YearlyReportBucket[];

// Top-N + Other truncation. Preserves the 'Unknown' bucket as a real entry regardless of rank.
// Tiebreak: count DESC, key ASC. Uses 10 for nationality per D-03.
export function truncateTopN(buckets: YearlyReportBucket[], n: number): YearlyReportBucket[];

// Decade fill: { md5, publication_year: number | null }[] -> [{ key: '1990s', count: N }, ..., { key: 'Unknown', count: U }].
// Zero-fills gaps between the min and max known decade.
export function fillDecades(rows: Array<{ md5: string; publication_year: number | null }>): YearlyReportBucket[];

// Coverage: passthrough from repository.getCoverageCounts shaped into the response block.
export function shapeCoverage(counts: ReturnTypeOfRepoGetCoverageCounts): YearlyReport['coverage'];
```

Composition for `getYearly(year)`:
1. Read `appConfig.reports.timeZone`.
2. Call `yearBoundsInZone(year, tz)` -> `{ startSec, endSec }`.
3. Repo: `getBooksReadInYear(startSec, endSec)` -> md5 list. (May be empty.)
4. Repo: `getReadingTotalsInYear(startSec, endSec)` -> totals (always include all reading).
5. Repo: per-breakdown queries scoped to that md5 list.
6. Service: `bucketWithUnknown` per breakdown (genre uses per-row, others per-book), `truncateTopN(natRows, 10)`, `fillDecades(yearRows)`.
7. Repo: `getCoverageCounts(md5s)` -> shape into response.
8. Return `YearlyReport`.

Edge case: empty md5 list -> per-breakdown queries return `[]`; bucketWithUnknown over `[]` returns `[]` (NOT a bare `{key:'Unknown', count:0}` row). Coverage all zeros. Totals from step 4 still reflect any reading (page-time totals include incomplete reading per Nyquist 3, even if the book isn't "read").
