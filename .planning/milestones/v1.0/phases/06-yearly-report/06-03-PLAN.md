```typescript
import { db } from '../knex';

// page_stat.start_time is Unix epoch SECONDS (KOReader native unit).
// All boundary parameters are seconds. Do NOT replicate the * 1000 mapping in StatsRepository.

export async function getYearsWithReading(): Promise<number[]>;

// Returns book.md5 set satisfying the >=95% predicate AND at least one page_stat row in [start, end).
export async function getBooksReadInYear(yearStartSec: number, yearEndSec: number): Promise<string[]>;

export async function getReadingTotalsInYear(yearStartSec: number, yearEndSec: number): Promise<{
  totalReadTimeSec: number;
  totalPageTurns: number;
}>;

// For each breakdown, returns rows of { md5, value } where value is nullable; the service
// will bucket NULLs into 'Unknown'. md5 is constrained to the books-read-in-year set, which
// the caller pre-computed.
export async function getGenresForBooks(md5s: string[]): Promise<Array<{ md5: string; genre: string | null }>>;
export async function getPrimaryAuthorNationalities(md5s: string[]): Promise<Array<{ md5: string; nationality: string | null }>>;
export async function getPublicationYears(md5s: string[]): Promise<Array<{ md5: string; publication_year: number | null }>>;
export async function getOriginalLanguages(md5s: string[]): Promise<Array<{ md5: string; original_language: string | null }>>;

// Coverage denominators: how many books-read-in-year have each field populated.
export async function getCoverageCounts(md5s: string[]): Promise<{
  total_books: number;
  genre_known: number;            // distinct md5s with at least one book_genre row
  nationality_known: number;      // distinct md5s where the position=0 author has nationality NOT NULL
  publication_year_known: number; // distinct md5s with publication_year NOT NULL
  original_language_known: number;// distinct md5s with original_language NOT NULL
}>;
```

Key SQL shapes (from RESEARCH Pattern 3):
- `getYearsWithReading`: `SELECT DISTINCT CAST(strftime('%Y', start_time, 'unixepoch') AS INTEGER) AS y FROM page_stat ORDER BY y DESC`. Keep it UTC-based (the years selector is a coarse picker; if a TZ-correct list is required later, the service can post-filter). Document this trade-off in a comment.
- `getBooksReadInYear`: a single CTE `WITH max_page_by_end AS (SELECT book_md5, MAX(page) AS max_p FROM page_stat WHERE start_time < :yearEnd GROUP BY book_md5)`, joined with `book` on md5; predicate `max_p >= CAST(0.95 * COALESCE(b.reference_pages, b.pages) AS INTEGER) AND COALESCE(b.reference_pages, b.pages) > 0 AND b.soft_deleted = 0 AND EXISTS (SELECT 1 FROM page_stat ps2 WHERE ps2.book_md5 = b.md5 AND ps2.start_time >= :yearStart AND ps2.start_time < :yearEnd)`.
- `getReadingTotalsInYear`: `SELECT COALESCE(SUM(duration), 0) AS time_sec, COUNT(*) AS turns FROM page_stat WHERE start_time >= :yearStart AND start_time < :yearEnd`. Page turns chosen for `total_pages` per RESEARCH Open Question 3 + matches `StatsService.totalPagesRead` precedent.
- Nationality query: `book_author.position = 0` filter (D-07).
- Soft-delete: every join on `book` includes `where('book.soft_deleted', false)`.

Tie-break rule: any `ORDER BY count DESC` query also tiebreaks by the textual key ascending (Pitfall 8), but most ordering is done in the service over JS arrays; if SQL emits a sorted list, append `, key ASC`.
