SWR hooks (mirror `apps/web/src/api/enrichment.ts` exactly):

```typescript
import useSWR from 'swr';
import { fetchFromAPI } from './api';
import type { YearlyReport, YearsResponse } from '@koinsight/common/dist/types/reports-api.js';

export function useReportYears() {
  return useSWR<YearsResponse>(
    'reports/years',
    () => fetchFromAPI<YearsResponse>('reports/years')
  );
}

export function useReportYearly(year: number | null) {
  return useSWR<YearlyReport>(
    year ? ['reports/yearly', year] : null,
    () => fetchFromAPI<YearlyReport>('reports/yearly', 'GET', { year: year! })
  );
}
```
