    D-03 (split): Split on regex /\s*(?:&|;|,|\band\b)\s*/i. Trim each segment. Collapse internal whitespace via .replace(/\s+/g, ' '). Periods are NOT separators. `and` matches whole word only.

    D-05 (suffix merge — runs FIRST, before D-04 flip): Suffix whitelist (case-insensitive, trailing period optional): {Jr., Sr., II, III, IV, PhD, MD}. After comma-splitting, if any non-first segment matches the suffix whitelist, merge it back onto the preceding segment with a single space. Example: ['Strunk', 'Jr.', 'William'] → ['Strunk Jr.', 'William'].

    D-04 (LN-FN flip — runs AFTER D-05 merge): If the ORIGINAL input contains ONLY commas as separators (no `&`, `;`, or whole-word `and`) AND post-merge segment count is exactly 2, flip to 'First Last' form by joining as `${segments[1]} ${segments[0]}`. Examples: 'Strunk, William' → 'William Strunk'; 'Tolkien, J.R.R.' → 'J.R.R. Tolkien'; 'Strunk, Jr., William' → 'William Strunk Jr.'. Strings with any other separator are NEVER flipped.

    D-06 (suspicious drop): After all above, drop any segment that is empty, whitespace-only, a single non-letter character, or pure punctuation (no letters at all). Remaining authors get contiguous position 0,1,2,... with no gaps.

    D-07 (determinism): Pure function. No locale-dependent casing, no Unicode folding, no external data. Null/undefined/empty input returns [].

    export type ParsedAuthor = {
      name: string;
      position: number;
    };

    const SUFFIX_WHITELIST = ['jr', 'sr', 'ii', 'iii', 'iv', 'phd', 'md'];
    const SEPARATOR_RE = /\s*(?:&|;|,|\band\b)\s*/i;
    const COMMA_ONLY_RE = /^[^&;]*$/; // no & or ;
    const HAS_AND_RE = /\band\b/i;
    const PUNCT_OR_SINGLE_NONLETTER_RE = /^[^A-Za-z]*$/; // no letters at all → drop

    function isSuffix(segment: string): boolean {
      const stripped = segment.trim().replace(/\.$/, '').toLowerCase();
      return SUFFIX_WHITELIST.includes(stripped);
    }

    export function parseAuthors(input: string | null | undefined): ParsedAuthor[] {
      if (input == null) return [];
      const original = input;
      const segments = original
        .split(SEPARATOR_RE)
        .map((s) => s.trim().replace(/\s+/g, ' '))
        .filter((s) => s.length > 0);

      // D-05: suffix merge (runs before flip)
      const merged: string[] = [];
      for (const seg of segments) {
        if (merged.length > 0 && isSuffix(seg)) {
          merged[merged.length - 1] = `${merged[merged.length - 1]} ${seg}`;
        } else {
          merged.push(seg);
        }
      }

      // D-04: LN-FN flip when ORIGINAL had only commas as separators AND we end up with exactly 2 segments
      const onlyCommas = COMMA_ONLY_RE.test(original) && !HAS_AND_RE.test(original);
      let finalNames = merged;
      if (onlyCommas && merged.length === 2) {
        finalNames = [`${merged[1]} ${merged[0]}`];
      }

      // D-06: drop suspicious segments (no letters)
      const cleaned = finalNames.filter((s) => !PUNCT_OR_SINGLE_NONLETTER_RE.test(s));

      return cleaned.map((name, position) => ({ name, position }));
    }
