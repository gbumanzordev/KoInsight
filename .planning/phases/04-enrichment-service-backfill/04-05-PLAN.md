     ```typescript
     import type { Knex } from 'knex';
     import { ENRICHMENT_POLL_INTERVAL_MS, ENRICHMENT_MAX_ATTEMPTS } from './constants';
     import { openLibraryClient } from '../open-library/open-library-client';
     import { wikidataClient } from './wikidata/wikidata-client';
     import { matchWork } from './matcher';
     import { classifyFailure, computeNextAttemptAt, truncateError } from './retry';
     import { applyEnrichment, markTerminalFailure, type EnrichedBundle } from './applier';

     export interface EnrichmentWorker { stop(): Promise<void>; }

     export function startEnrichmentWorker(knex: Knex): EnrichmentWorker {
       let isShuttingDown = false;
       let timerHandle: NodeJS.Timeout | null = null;
       let currentJob: Promise<void> | null = null;

       // D-05 crash recovery (synchronous-ish; we return after it completes)
       const readyPromise = knex('enrichment_job').where({ status: 'running' }).update({ status: 'pending' })
         .then((n) => { if (n > 0) console.log(`enrichment worker: reset ${n} running -> pending`); });

       async function tick() {
         if (isShuttingDown) return;
         try {
           currentJob = claimAndProcess();
           await currentJob;
         } catch (err) {
           console.error('enrichment worker tick failed', err);
         } finally {
           currentJob = null;
         }
         if (!isShuttingDown) timerHandle = setTimeout(tick, ENRICHMENT_POLL_INTERVAL_MS);
       }

       async function claimAndProcess(): Promise<void> {
         const result = await knex.raw(`
           UPDATE enrichment_job
           SET status = 'running', attempts = attempts + 1, updated_at = CURRENT_TIMESTAMP
           WHERE id = (
             SELECT id FROM enrichment_job
             WHERE status = 'pending'
               AND (next_attempt_at IS NULL OR next_attempt_at <= CURRENT_TIMESTAMP)
             ORDER BY created_at ASC
             LIMIT 1
           )
           RETURNING *
         `);
         const rows = Array.isArray(result) ? result : (result?.rows ?? []);
         const job = rows[0];
         if (!job) return;
         try {
           await processJob(knex, job);
         } catch (err) {
           await scheduleRetryOrFail(knex, job, err);
         }
       }

       // Kick off: wait for ready then schedule first tick
       readyPromise.then(() => {
         if (!isShuttingDown) timerHandle = setTimeout(tick, ENRICHMENT_POLL_INTERVAL_MS);
       });

       return {
         async stop() {
           isShuttingDown = true;
           if (timerHandle) clearTimeout(timerHandle);
           if (currentJob) { try { await currentJob; } catch { /* ignored */ } }
         },
       };
     }

     async function processJob(knex: Knex, job: { id: number; book_md5: string }): Promise<void> {
       const book = await knex('book').where({ md5: job.book_md5 }).first();
       if (!book) throw new Error(`processJob: book ${job.book_md5} not found`);

       const primaryAuthor = (book.authors ?? '').split(',')[0]?.trim() ?? '';
       const search = await openLibraryClient.searchWork({ title: book.title, author: primaryAuthor });
       const candidate = matchWork({ title: book.title, authors: book.authors }, search.docs ?? []);
       if (!candidate) {
         await markTerminalFailure(knex, job.id, job.book_md5, new Error('no-match after top-3 candidates'));
         return;
       }

       const editionKey = candidate.cover_edition_key ?? (candidate as { edition_key?: string[] }).edition_key?.[0];
       const edition = editionKey ? await openLibraryClient.getEdition(editionKey) : null;
       const workKey = edition?.works?.[0]?.key ?? candidate.key;
       if (!workKey) throw new Error('no work key derivable from OL candidate');
       const work = await openLibraryClient.getWork(workKey);

       // Resolve authors + nationalities
       const enrichedAuthors: EnrichedBundle['authors'] = [];
       for (const a of work.authors ?? []) {
         const authorKey = (a as { author?: { key?: string } }).author?.key;
         if (!authorKey) continue;
         const author = await openLibraryClient.getAuthor(authorKey);
         const wikidataQid = (author as { remote_ids?: { wikidata?: string } }).remote_ids?.wikidata;
         const nationality = wikidataQid ? await wikidataClient.resolveP27Nationality(wikidataQid) : null;
         enrichedAuthors.push({ name: author.name, openlibrary_key: authorKey, nationality });
       }

       const bundle: EnrichedBundle = {
         workKey,
         publicationYear: extractPublicationYear(work, edition),
         originalLanguage: extractOriginalLanguage(work, edition),
         authors: enrichedAuthors,
         subjects: work.subjects ?? [],
       };

       await applyEnrichment(knex, job.book_md5, job.id, bundle);
     }

     async function scheduleRetryOrFail(knex: Knex, job: { id: number; book_md5: string; attempts: number }, err: unknown): Promise<void> {
       const klass = classifyFailure(err);
       if (klass === 'permanent') {
         await markTerminalFailure(knex, job.id, job.book_md5, err);
         return;
       }
       if (job.attempts >= ENRICHMENT_MAX_ATTEMPTS) {
         await markTerminalFailure(knex, job.id, job.book_md5, err);
         return;
       }
       await knex('enrichment_job').where({ id: job.id }).update({
         status: 'pending',
         last_error: truncateError(String(err instanceof Error ? err.message : err)),
         next_attempt_at: computeNextAttemptAt(job.attempts, new Date()),
         updated_at: knex.fn.now(),
       });
     }

     function extractPublicationYear(work: unknown, edition: unknown): number | null {
       // Prefer edition.publish_date parsed as year; fallback to work.first_publish_date. Return null if unparseable.
       const raw = (edition as { publish_date?: string })?.publish_date ?? (work as { first_publish_date?: string })?.first_publish_date;
       if (!raw) return null;
       const match = String(raw).match(/\b(\d{4})\b/);
       return match ? Number(match[1]) : null;
     }

     function extractOriginalLanguage(work: unknown, edition: unknown): string | null {
       // OL work.original_languages[0].key = '/languages/eng' -> 'en'. Fallback to edition.languages[0].
       const raw = (work as { original_languages?: Array<{ key?: string }> })?.original_languages?.[0]?.key
         ?? (edition as { languages?: Array<{ key?: string }> })?.languages?.[0]?.key;
       if (!raw) return null;
       const code3 = raw.replace('/languages/', '');
       return mapIso6392to6391(code3);
     }

     function mapIso6392to6391(code3: string): string | null {
       // Minimal map; Phase 3 country-codes module has the canonical approach. Stub here; could be replaced with a shared module in a future phase.
       const m: Record<string, string> = { eng: 'en', spa: 'es', fre: 'fr', fra: 'fr', ger: 'de', deu: 'de', rus: 'ru', jpn: 'ja', chi: 'zh', zho: 'zh' };
       return m[code3] ?? null;
     }
     ```
