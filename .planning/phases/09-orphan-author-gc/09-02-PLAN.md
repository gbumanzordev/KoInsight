    ```typescript
    import { Router, Request, Response } from 'express';
    import { z } from 'zod';
    import { db } from '../knex';
    import { deleteOrphanAuthors } from './orphan-author-gc';

    // D-07: body schema. `confirm` must be the exact literal 'DELETE_ORPHANS' (case-sensitive).
    const orphanAuthorGcBodySchema = z.object({
      confirm: z.literal('DELETE_ORPHANS'),
      dry_run: z.boolean().optional(),
    });

    const router = Router();

    /**
     * POST /api/admin/authors/gc
     * Phase 9 (AUTHGC-01, AUTHGC-03):
     *   - D-07: body validated by Zod; literal `confirm: 'DELETE_ORPHANS'` required.
     *   - D-08: only POST registered. Other methods naturally return 404 (Express default).
     *   - D-09: response is { deleted, dry_run, sample }; sample is [] when deleted is 0.
     *   - D-10: console.info on successful non-dry-run; console.error on caught exception (-> 500).
     *   - dry_run resolution: explicit body field wins; otherwise query ?dry_run=1 (or '1'/'true') -> true.
     */
    router.post('/authors/gc', async (req: Request, res: Response) => {
      const parsed = orphanAuthorGcBodySchema.safeParse(req.body);
      if (!parsed.success) {
        res.status(400).json({ error: parsed.error.flatten() });
        return;
      }

      const queryDryRun = req.query.dry_run === '1' || req.query.dry_run === 'true';
      const dryRun =
        parsed.data.dry_run !== undefined ? parsed.data.dry_run : queryDryRun;

      try {
        const result = await deleteOrphanAuthors(db, { dryRun });
        if (!dryRun) {
          console.info('admin:orphan-author-gc', { deleted: result.deleted, sample: result.sample });
        }
        res.status(200).json({
          deleted: result.deleted,
          dry_run: dryRun,
          sample: result.sample,
        });
      } catch (error) {
        console.error('admin:orphan-author-gc failed', error);
        res.status(500).json({ error: 'Failed to delete orphan authors' });
      }
    });

    export { router as adminRouter };
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Internet -> Express (POST /api/admin/authors/gc) | Untrusted input crosses here. CORS is currently `origin: '*'` (app.ts:30) — any web origin can issue cross-origin POSTs once the user is on a network with reach to the server. |
| Express handler -> deleteOrphanAuthors | Trusted boundary; handler validates body via Zod before invoking the core. |
| Express middleware seam -> future auth | The `app.use('/api/admin', adminRouter)` mount line is the wrap-point a future auth middleware will sit in front of (`app.use('/api/admin', requireAdmin, adminRouter)`). No rework expected. |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-06 | Spoofing | Anonymous caller deletes data | accept (deferred) | Real auth lands in next milestone (CONTEXT deferred-ideas). For Phase 9 the literal `confirm: 'DELETE_ORPHANS'` body string is sufficient to defeat accidental triggering — explicit acceptance per SPEC §Boundaries. The mount line shape `app.use('/api/admin', adminRouter)` lets future auth wrap as middleware without changing route code. |
| T-09-07 | Tampering | Stray browser navigation triggers deletion | mitigate | Endpoint is POST-only with required JSON body (D-08); GET / DELETE / etc. return 404. Test cases 5-6 in Task 2 lock this behavior. A `<a href>` click cannot produce a JSON-body POST. |
| T-09-08 | Tampering | Confirm-string bypass via case-fold or empty string | mitigate | `z.literal('DELETE_ORPHANS')` is case-sensitive; missing field also fails parse. Test cases 2-4 in Task 2 lock this. |
| T-09-09 | Tampering | CSRF from a logged-in browser session | accept (deferred) | No session cookies exist today (no auth). When auth ships, the auth middleware should add a CSRF defense (e.g., custom header, SameSite cookie, or CORS origin allow-list). Out of scope for Phase 9 per SPEC. |

| T-09-13 | Elevation of Privilege | Caller exploits bug in another /api/admin/* endpoint | mitigate | `adminRouter` only registers POST /authors/gc this phase; no other routes are exposed. Future siblings will be reviewed at the time they ship. |
