    ```typescript
    export type AuthorRow = {
      id: number;
      name: string;
      openlibrary_key: string | null;
      wikidata_qid: string | null;
      nationality: string | null;
      nationality_source: 'openlibrary' | 'manual' | null;

    type FakeAuthor = Omit<AuthorRow, 'id' | 'created_at' | 'updated_at'>;

    export function fakeAuthor(overrides: Partial<FakeAuthor> = {}): FakeAuthor {
      return {
        name: faker.person.fullName(),
        openlibrary_key: null,
        wikidata_qid: null,
        nationality: null,
        nationality_source: null,
        bio: null,
        ...overrides,
      };
    }

    export async function createAuthor(
      db: Knex,
      overrides: Partial<FakeAuthor> = {}
    ): Promise<AuthorRow> {
      const data = fakeAuthor(overrides);
      const [row] = await db<AuthorRow>('author').insert(data).returning('*');
      return row;
    }
    ```

    ```typescript
    export type OrphanAuthorGcResult = {
    ```

    Implementation requirements:
    1. Predicate is exactly (D-06): `SELECT id, name FROM author WHERE id NOT IN (SELECT DISTINCT author_id FROM book_author)`. Implement via Knex builder, not a raw string, so SQLite escaping is handled by the driver:
       ```typescript
       const orphans = await runner<{ id: number; name: string }>('author')
         .select('id', 'name')
         .whereNotIn('id', runner('book_author').distinct('author_id'));
       ```
       where `runner` is either `db` (dry-run) or the transaction `trx` (delete path).
    2. `sample` is the first 20 captured rows by query order (D-06): `const sample = orphans.slice(0, 20);`. `sample` is `[]` when `orphans.length === 0` (slice naturally produces `[]`).
    3. Dry-run path (D-05): no transaction; perform the SELECT only; return `{ deleted: orphans.length, sample }`.
    4. Delete path (D-05): wrap in `db.transaction(async (trx) => { ... })`. Inside the transaction:
       - Run the predicate SELECT against `trx` (so the read and write see the same snapshot).
       - If `orphans.length === 0`, return `{ deleted: 0, sample: [] }` from the transaction (no DELETE issued).
       - Otherwise: `await trx('author').whereIn('id', orphans.map(o => o.id)).delete();`
       - Return `{ deleted: orphans.length, sample }` from the transaction body.
    5. The function must NOT import `db` from `../knex` — only `import type { Knex } from 'knex'`. This makes per-test Knex injection trivial (callers always pass `db`).
    6. Add a JSDoc header at the top of the file naming the SPEC requirements covered (AUTHGC-01, AUTHGC-02, AUTHGC-03) and the predicate used; mirror the comment style of `backfill-reference-pages.ts` (Phase + decision IDs).

    Anti-patterns (do NOT do):
    - Do NOT use `db.raw('NOT IN ...')` — Knex builder is sufficient and avoids hand-rolled SQL injection surface.
    - Do NOT import `db` from `../knex` (breaks per-test injection).
    - Do NOT iterate per-row deletes — use a single `whereIn(...).delete()` so it is one DML statement.
    - Do NOT log inside this function — logging belongs in the route handler (D-10) and CLI (D-13).
